001package org.xbib.elasticsearch.index.mapper.langdetect;
002
003import org.elasticsearch.common.xcontent.XContentBuilder;
004import org.elasticsearch.common.xcontent.XContentParser;
005import org.elasticsearch.index.mapper.FieldMapperListener;
006import org.elasticsearch.index.mapper.Mapper;
007import org.elasticsearch.index.mapper.MapperParsingException;
008import org.elasticsearch.index.mapper.MergeContext;
009import org.elasticsearch.index.mapper.MergeMappingException;
010import org.elasticsearch.index.mapper.ObjectMapperListener;
011import org.elasticsearch.index.mapper.ParseContext;
012import org.elasticsearch.index.mapper.core.StringFieldMapper;
013import org.xbib.elasticsearch.common.langdetect.Detector;
014import org.xbib.elasticsearch.common.langdetect.Language;
015import org.xbib.elasticsearch.common.langdetect.LanguageDetectionException;
016
017import java.io.IOException;
018import java.nio.charset.Charset;
019import java.util.List;
020import java.util.Map;
021
022import static org.elasticsearch.index.mapper.MapperBuilders.stringField;
023
024public class LangdetectMapper implements Mapper {
025
026    public static final String CONTENT_TYPE = "langdetect";
027
028    public static class Builder extends Mapper.Builder<Builder, LangdetectMapper> {
029
030        private StringFieldMapper.Builder contentBuilder;
031        private StringFieldMapper.Builder langBuilder = stringField("lang");
032        private Detector detector;
033
034        public Builder(String name, Detector detector) {
035            super(name);
036            this.detector = detector;
037            this.contentBuilder = stringField(name);
038            this.builder = this;
039        }
040
041        public Builder content(StringFieldMapper.Builder content) {
042            this.contentBuilder = content;
043            return this;
044        }
045
046        public Builder lang(StringFieldMapper.Builder lang) {
047            this.langBuilder = lang;
048            return this;
049        }
050
051        @Override
052        public LangdetectMapper build(BuilderContext context) {
053            context.path().add(name);
054            StringFieldMapper contentMapper = contentBuilder.build(context);
055            StringFieldMapper langMapper = langBuilder.build(context);
056            context.path().remove();
057            return new LangdetectMapper(name, detector, contentMapper, langMapper);
058        }
059    }
060
061    public static class TypeParser implements Mapper.TypeParser {
062
063        private Detector detector;
064
065        public TypeParser(Detector detector) {
066            this.detector = detector;
067        }
068
069        @SuppressWarnings({"unchecked", "rawtypes"})
070        @Override
071        public Mapper.Builder parse(String name, Map<String, Object> node, ParserContext parserContext)
072                throws MapperParsingException {
073            LangdetectMapper.Builder builder = new Builder(name, detector);
074            for (Map.Entry<String, Object> entry : node.entrySet()) {
075                String fieldName = entry.getKey();
076                Object fieldNode = entry.getValue();
077
078                if (fieldName.equals("fields")) {
079                    Map<String, Object> fieldsNode = (Map<String, Object>) fieldNode;
080                    for (Map.Entry<String, Object> fieldsEntry : fieldsNode.entrySet()) {
081                        String propName = fieldsEntry.getKey();
082                        Object propNode = fieldsEntry.getValue();
083
084                        if (name.equals(propName)) {
085                            builder.content((StringFieldMapper.Builder) parserContext.typeParser("string").parse(name,
086                                    (Map<String, Object>) propNode, parserContext));
087                        } else if ("lang".equals(propName)) {
088                            builder.lang((StringFieldMapper.Builder) parserContext.typeParser("string").parse("lang",
089                                    (Map<String, Object>) propNode, parserContext));
090                        }
091                    }
092                }
093            }
094
095            return builder;
096        }
097    }
098
099    private final String name;
100    private final Detector detector;
101    private final StringFieldMapper contentMapper;
102    private final StringFieldMapper langMapper;
103
104    public LangdetectMapper(String name, Detector detector, StringFieldMapper contentMapper, StringFieldMapper langMapper) {
105        this.name = name;
106        this.detector = detector;
107        this.contentMapper = contentMapper;
108        this.langMapper = langMapper;
109    }
110
111    @Override
112    public String name() {
113        return name;
114    }
115
116    @Override
117    public void parse(ParseContext context) throws IOException {
118        String content = null;
119
120        XContentParser parser = context.parser();
121        XContentParser.Token token = parser.currentToken();
122
123        if (token == XContentParser.Token.VALUE_STRING) {
124            // try decode UTF-8 base64 (e.g. from attachment mapper plugin)
125            content = parser.text();
126            try {
127                byte[] b = parser.binaryValue();
128                if (b != null && b.length > 0) {
129                    content = new String(b, Charset.forName("UTF-8"));
130                }
131            } catch (Exception e) {
132                // ignore
133            }
134        }
135
136        context.externalValue(content);
137        contentMapper.parse(context);
138
139        if (content == null) {
140            return;
141        }
142
143        try {
144            List<Language> langs = detector.detectAll(content);
145            for (Language lang : langs) {
146                context.externalValue(lang.getLanguage());
147                langMapper.parse(context);
148            }
149        } catch (LanguageDetectionException e) {
150            context.externalValue("unknown");
151            langMapper.parse(context);
152        }
153    }
154
155    @Override
156    public void merge(Mapper mergeWith, MergeContext mergeContext) throws MergeMappingException {
157    }
158
159    @Override
160    public void traverse(FieldMapperListener fieldMapperListener) {
161        contentMapper.traverse(fieldMapperListener);
162        langMapper.traverse(fieldMapperListener);
163    }
164
165    @Override
166    public void traverse(ObjectMapperListener objectMapperListener) {
167    }
168
169    @Override
170    public void close() {
171        contentMapper.close();
172        langMapper.close();
173    }
174
175    @Override
176    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
177        builder.startObject(name);
178        builder.field("type", CONTENT_TYPE);
179
180        builder.startObject("fields");
181        contentMapper.toXContent(builder, params);
182        langMapper.toXContent(builder, params);
183        builder.endObject();
184
185        builder.endObject();
186        return builder;
187    }
188}

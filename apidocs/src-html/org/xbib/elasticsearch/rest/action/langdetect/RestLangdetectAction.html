001package org.xbib.elasticsearch.rest.action.langdetect;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.settings.Settings;
006import org.elasticsearch.rest.BaseRestHandler;
007import org.elasticsearch.rest.RestChannel;
008import org.elasticsearch.rest.RestController;
009import org.elasticsearch.rest.RestRequest;
010import org.elasticsearch.rest.action.support.RestStatusToXContentListener;
011import org.xbib.elasticsearch.action.langdetect.LangdetectAction;
012import org.xbib.elasticsearch.action.langdetect.LangdetectRequest;
013import org.xbib.elasticsearch.action.langdetect.LangdetectResponse;
014
015import static org.elasticsearch.rest.RestRequest.Method.POST;
016
017public class RestLangdetectAction extends BaseRestHandler {
018
019    @Inject
020    public RestLangdetectAction(Settings settings, Client client, RestController controller) {
021        super(settings, client);
022        controller.registerHandler(POST, "/_langdetect", this);
023    }
024
025    @Override
026    public void handleRequest(final RestRequest request, final RestChannel channel) {
027        LangdetectRequest langdetectRequest = new LangdetectRequest().setText(request.content());
028        client.execute(LangdetectAction.INSTANCE, langdetectRequest, new RestStatusToXContentListener<LangdetectResponse>(channel));
029
030    }
031}

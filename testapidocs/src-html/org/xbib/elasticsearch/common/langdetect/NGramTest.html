065    /**
066     * Test method for {@link NGram#get(int)} and {@link NGram#addChar(char)}
067     */
068    @Test
069    public final void testNGram() {
070        NGram ngram = new NGram();
071        assertEquals(ngram.get(0), null);
072        assertEquals(ngram.get(1), null);
073        assertEquals(ngram.get(2), null);
074        assertEquals(ngram.get(3), null);
075        assertEquals(ngram.get(4), null);
076        ngram.addChar(' ');
077        assertEquals(ngram.get(1), null);
078        assertEquals(ngram.get(2), null);
079        assertEquals(ngram.get(3), null);
080        ngram.addChar('A');
081        assertEquals(ngram.get(1), "A");
082        assertEquals(ngram.get(2), " A");
083        assertEquals(ngram.get(3), null);
084        ngram.addChar('\u06cc');
085        assertEquals(ngram.get(1), "\u064a");
086        assertEquals(ngram.get(2), "A\u064a");
087        assertEquals(ngram.get(3), " A\u064a");
088        ngram.addChar('\u1ea0');
089        assertEquals(ngram.get(1), "\u1ec3");
090        assertEquals(ngram.get(2), "\u064a\u1ec3");
091        assertEquals(ngram.get(3), "A\u064a\u1ec3");
092        ngram.addChar('\u3044');
093        assertEquals(ngram.get(1), "\u3042");
094        assertEquals(ngram.get(2), "\u1ec3\u3042");
095        assertEquals(ngram.get(3), "\u064a\u1ec3\u3042");
096
097        ngram.addChar('\u30a4');
098        assertEquals(ngram.get(1), "\u30a2");
099        assertEquals(ngram.get(2), "\u3042\u30a2");
100        assertEquals(ngram.get(3), "\u1ec3\u3042\u30a2");
101        ngram.addChar('\u3106');
102        assertEquals(ngram.get(1), "\u3105");
103        assertEquals(ngram.get(2), "\u30a2\u3105");
104        assertEquals(ngram.get(3), "\u3042\u30a2\u3105");
105        ngram.addChar('\uac01');
106        assertEquals(ngram.get(1), "\uac00");
107        assertEquals(ngram.get(2), "\u3105\uac00");
108        assertEquals(ngram.get(3), "\u30a2\u3105\uac00");
109        ngram.addChar('\u2010');
110        assertEquals(ngram.get(1), null);
111        assertEquals(ngram.get(2), "\uac00 ");
112        assertEquals(ngram.get(3), "\u3105\uac00 ");
113
114        ngram.addChar('a');
115        assertEquals(ngram.get(1), "a");
116        assertEquals(ngram.get(2), " a");
117        assertEquals(ngram.get(3), null);
118
119    }
